# Run the Qt deployment tool to copy DLLs and translation files
#
# param target the target when we want to copy (like the executable name)
# optional parameter: the path to the output dir where the DLL will be copied
# 	default is the target binary dir
# optional paramter: the path to the executable we want to run the Qt deployment tool on
# 	default is the executable generated by the given target.
# 	Note: the previous optional parameter must be set to set this one
function(deployqt target)
	set (outputdir "$<TARGET_FILE_DIR:${target}>")
	set (executable "$<TARGET_FILE:${target}>")
	if (WIN32)
		set (deployqtexe "${QT_ROOT_PATH}/bin/windeployqt.exe")
	elseif (APPLE)
		set (deployqtexe "${QT_ROOT_PATH}/bin/macdeployqt")
	else()
		return()
	endif()
	if (${ARGC} GREATER 1)
		set (outputdir ${ARGV1})
	endif()
	if (${ARGC} GREATER 2)
		set (executable ${ARGV2})
	endif()

	add_custom_command(TARGET ${target} POST_BUILD
		COMMAND "${deployqtexe}"
			$<$<CONFIG:debug>:--debug>$<$<CONFIG:release>:--release>
			--verbose 0
			--no-compiler-runtime
			--dir "${outputdir}"
			"${executable}"
		COMMENT "Deploying Qt..."
	)
endfunction(deployqt)

# Run QLALR
#
# set_qlalr(
#	CLASSNAME ClassName
#	DIRECTORY dir
#	[GENERATEDOT]
# )
#
# `ClassName` is the name of the class generated by QLALR
#	we assume we need to use these files:
#		- ClassName.g (input file)
#		- ClassName.hpp (generated)
#		- ClassName_gen.cpp (generated)
#		- table_classname_p.h (generated)
#		- classname.cpp (generated)
# `dir` is the directory where the input and output files are
# `GENERATEDOT` if specified, it generates dot SVG file
function(set_qlalr)
	set(options GENERATEDOT)
	set(oneValueArgs CLASSNAME DIRECTORY)
	cmake_parse_arguments(QLALR "${options}" "${oneValueArgs}" "" ${ARGN})
	string(TOLOWER ${QLALR_CLASSNAME} QLALR_CLASSNAME_LOWER)

	find_program(QLALR_EXE
		NAMES "qlalr" "qlalr.exe"
		HINTS "${QT_ROOT_PATH}"
		PATH_SUFFIXES "bin"
		DOC "QLALR executable path"
	)
	find_program(DOT_EXE
		NAMES "dot" "dot.exe"
		PATH_SUFFIXES "bin"
		DOC "Graphviz Dot executable path"
	)
	if(QLALR_EXE)
		set(DOT_OPTION "")
		set(DOT_COMMAND "")
		if(DOT_EXE AND QLALR_GENERATEDOT)
			set(DOT_OPTION "--dot")
			set(DOT_COMMAND "> ${QLALR_CLASSNAME}.gv && \"${DOT_EXE}\" -Tsvg ${QLALR_CLASSNAME}.gv -o ${QLALR_CLASSNAME}.svg")
		endif()
		add_custom_command(
			OUTPUT
				"${QLALR_DIRECTORY}/${QLALR_CLASSNAME}.hpp"
				"${QLALR_DIRECTORY}/${QLALR_CLASSNAME}_gen.cpp"
				"${QLALR_DIRECTORY}/table_${QLALR_CLASSNAME_LOWER}.cpp"
				"${QLALR_DIRECTORY}/table_${QLALR_CLASSNAME_LOWER}_p.h"
			COMMAND "${QLALR_EXE}" --no-debug --no-lines ${DOT_OPTION} "${QLALR_CLASSNAME}.g" ${DOT_COMMAND}
			MAIN_DEPENDENCY "${QLALR_DIRECTORY}/${QLALR_CLASSNAME}.g"
			WORKING_DIRECTORY "${QLALR_DIRECTORY}"
			COMMENT "Running ${QLALR_EXE} for ${QLALR_CLASSNAME}.g"
		)
	endif()
endfunction(set_qlalr)

# Copy files in the build directory at build time if files don't already exist.
#
# param target the target when we want to copy (like the executable name)
# param debug_files (list) files that should go into the debug folder
# param release_files (list) files that should go into the release folder (can be the same as debug_files)
# param subdir an optionnal subdir where to write the files
# param (optional) forced - tells if we should copy even if the files already exist
#
# Note: `debug_files` and `release_files` must have the same length
# Note: this function should be called after the definition of the executable
function(post_build_copy_dlls target debug_files release_files subdir)
	string (REPLACE ";" "\" \"" debug_files_str "${debug_files}")
	string (REPLACE ";" "\" \"" release_files_str "${release_files}")
	add_custom_command(TARGET ${target} POST_BUILD
		COMMAND "${CMAKE_COMMAND}" -E make_directory
			"$<TARGET_FILE_DIR:${target}>/${subdir}"
		COMMENT "Creating subdir ${subdir}"
	)
	if (${ARGC} GREATER 1)
		set (_command "copy")
	else()
		set (_command "copy_if_different")
	endif()
	add_custom_command(TARGET ${target} POST_BUILD
		COMMAND "${CMAKE_COMMAND}" -E ${_command}
			$<$<CONFIG:debug>:"${debug_files_str}">$<$<CONFIG:release>:"${release_files_str}">
			"$<TARGET_FILE_DIR:${target}>/${subdir}"
		COMMENT "Copying dlls to output directory"
	)
endfunction(post_build_copy_dlls)

# Set the given project as the default startup project in MSVC++.
#
# param project_name the project that should be set as the default startup project
#
# Note: this function should be called after the definition of the executable
function(set_vs_default_startup_project project_name)
	set_property(DIRECTORY PROPERTY VS_STARTUP_PROJECT ${project_name})
endfunction(set_vs_default_startup_project)

function(_install_qt_dlls exe_name)
	if (WIN32)
		set (deployqtexe "${QT_ROOT_PATH}/bin/windeployqt.exe")
	elseif (APPLE)
		set (deployqtexe "${QT_ROOT_PATH}/bin/macdeployqt")
	endif()
	# Generate the Qt DLLs
	set(_bindir "${CMAKE_BINARY_DIR}")
	if(DEFINED CMAKE_RUNTIME_OUTPUT_DIRECTORY)
		set(_bindir "${CMAKE_RUNTIME_OUTPUT_DIRECTORY}")
	endif()
	install(CODE
		"
			if(\${BUILD_TYPE} STREQUAL \"Debug\")
				message(\"Deploying Debug Qt dlls for ${exe_name}...\")
				execute_process(
					COMMAND \"${deployqtexe}\"
					--debug
					--verbose 0
					--no-compiler-runtime
					--dir \"${CMAKE_BINARY_DIR}/qt_dlls\"
					\"${_bindir}/Debug/${exe_name}\"
				)
			else()
				message(\"Deploying Release Qt dlls for ${exe_name}...\")
				execute_process(
					COMMAND \"${deployqtexe}\"
					--release
					--verbose 0
					--no-compiler-runtime
					--dir \"${CMAKE_BINARY_DIR}/qt_dlls\"
					\"${_bindir}/Release/${exe_name}\"
				)
			endif()
		"
		COMPONENT libraries
	)
endfunction()

# Create an installer package.
#
# The installer contains the minimum required files:
# 	- the executable
# 	- the VC redist
# 	- if `USE_QT`, runs `windeployqt`
# 		- in this case, you can add optional parameters to binary files that we need to run
# 			windeployqt on it.
# 	- if `CREATE_DOC`, adds the doc
#
# NOTE: You can add extra files to the installer thanks to the function
# 		`installer_copy_files` (see below)
#
# Note: this function should be called after the definition of the executable
function(create_default_installer)
set(CPACK_MONOLITHIC_INSTALL FALSE)
	cpack_add_component(application
		DISPLAY_NAME "${CPACK_PACKAGE_NAME}"
		DESCRIPTION "Install ${CPACK_PACKAGE_DESCRIPTION_SUMMARY}"
		REQUIRED
		GROUP App
		INSTALL_TYPES default
	)
	cpack_add_component(libraries
		DISPLAY_NAME "Libraries"
		DESCRIPTION "Install required libraries to run the program."
		REQUIRED
		GROUP App
		INSTALL_TYPES default
	)
	cpack_add_component(documentation
		DISPLAY_NAME "Documentation"
		DESCRIPTION "Install some documentation in the install folder."
		GROUP Doc
		DEPENDS application
		INSTALL_TYPES default
	)


	if(UNIX)
		return()
	endif()

	install(
		TARGETS ${PROJECT_NAME}
		DESTINATION "."
		COMPONENT application
	)

	# Install VCRuntime
	if (MSVC)
		install(
			PROGRAMS "${MSVC_REDIST_DIR}/${VCREDIST_INSTALLER_NAME}"
			DESTINATION "."
			COMPONENT libraries
		)
	endif()

	# If Qt, we install the DLLs
	if (${USE_QT})
		# Remove the Qt folder
		install(
			CODE
			"file(REMOVE_RECURSE ${CMAKE_BINARY_DIR}/qt_dlls)"
			COMPONENT libraries
		)
		# Generate the Qt DLLs
		_install_qt_dlls("${PROJECT_NAME}.exe")
		foreach(b ${ARGN})
			_install_qt_dlls("${b}")
		endforeach()
		# Install the Qt DLLs
		install(
			DIRECTORY "${CMAKE_BINARY_DIR}/qt_dlls/"
			DESTINATION "."
			COMPONENT libraries
		)
	endif()
	
	# If OpenSSL
	if (OPENSSL_FOUND)
		# Remove
		install(
			CODE
			"file(REMOVE_RECURSE ${CMAKE_BINARY_DIR}/openssl_dlls)"
			COMPONENT libraries
		)
		# Install
		get_target_property(OPENSSL_DLLS OpenSSL::SSL RUNTIME_OBJECTS)
		install(
			FILES ${OPENSSL_DLLS}
			DESTINATION "."
			COMPONENT libraries
		)
	endif()

	# If we generate the doc, we install it
	if (${CREATE_DOC})
		install(
			DIRECTORY "${CMAKE_SOURCE_DIR}/../doc/"
			DESTINATION "./doc/"
			COMPONENT documentation
		)
	endif()
endfunction(create_default_installer)

# When creating the installer, this function can be called to add files in the installer.
#
# param files a list of files to add
# param destination the destination in the installation folder
#
# Note: this function should be called after the definition of the executable
# 		and preferably after the call to `create_default_installer`
function(installer_copy_files files destination)
	install(
		FILES ${files}
		DESTINATION "${destination}"
		COMPONENT application
	)
endfunction(installer_copy_files)

		COMMAND ${CMAKE_COMMAND} -E copy_directory

endfunction(copyReframeDLL)

	set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -pthread")
	add_compile_options(
		# generic
		"-Wall" "-Wextra" "-pedantic" "-ftemplate-backtrace-limit=0" # the last one is because of Eigen
	)
	# debug

		add_compile_options("-g" "-O0")
	# release
	else()
		add_compile_options("-O2")

		set(N 8)
	endif()
	add_compile_options(
		# generic
		"/Zc:__cplusplus" "/cgthreads${N}" "/MP" "/W4"
		# debug
		"$<$<CONFIG:debug>:/DEBUG>"
		"$<$<CONFIG:debug>:/Od>"
		# release
		"$<$<CONFIG:release>:/O2>"
		"$<$<CONFIG:release>:/GL>"

	add_link_options(
		# release
		"$<$<CONFIG:release>:/LTCG>"

	include(InstallRequiredSystemLibraries)
